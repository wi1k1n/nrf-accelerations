% Empty pages

\def\cleardoublepage{\clearpage%
    \if@twoside
    \ifodd\c@page\else
        \begin{minipage}[c][\textheight][c]{\textwidth}
            \begin{center}
                %This page intentionally left blank.
            \end{center}
        \end{minipage}

        \thispagestyle{plainline}

    \fi
    \fi
